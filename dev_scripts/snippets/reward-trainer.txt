MyMap_EventScript_RewardTrainer::
	goto_if_set FLAG_GOT_RICK_REWARD, MyMap_EventScript_RewardTrainer_PostBattleText
	trainerbattle_single TRAINER_BUG_CATCHER_RICK, MyMap_Text_TrainerIntro, MyMap_Text_TrainerDefeat, MyMap_EventScript_RewardTrainer_AfterBattle
	msgbox MyMap_Text_TrainerPostBattle, MSGBOX_AUTOCLOSE
	end

	giveitem ITEM_RARE_CANDY
	goto_if_eq VAR_RESULT, FALSE, Common_EventScript_BagIsFull
	setflag FLAG_GOT_RICK_REWARD
	msgbox MyMap_Text_TrainerRewardText, MSGBOX_DEFAULT
	release
	end
